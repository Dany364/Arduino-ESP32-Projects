COD SALVARE INFORMATIICARTELA CU PAROLA: #include <SPI.h>
#include <MFRC522.h>
#include <Wire.h>
#include <Keypad.h>
#include <EEPROM.h>

///////////////////////
// Configurare RC522 //
///////////////////////
#define SS_PIN 5
#define RST_PIN 4
MFRC522 mfrc522(SS_PIN, RST_PIN);

///////////////////////
// Configurare Keypad //
///////////////////////
const byte ROWS = 4; 
const byte COLS = 4; 

char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {32, 33, 25, 26}; 
byte colPins[COLS] = {27, 14, 12, 13}; 

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

///////////////////////
// Configurare EEPROM //
///////////////////////
#define EEPROM_SIZE 512  
#define PASSWORD_ADDR 500
#define PASSWORD_LENGTH 4

///////////////////////
// Configurare Buzzer //
///////////////////////
#define BUZZER_PIN 15

String enteredPassword = "";
bool passwordSet = false;
unsigned long lastZeroPress = 0;
bool zeroPressedOnce = false;

void setup() {
  Serial.begin(115200);
  pinMode(BUZZER_PIN, OUTPUT);

  SPI.begin();
  mfrc522.PCD_Init();
  Serial.println("=== Sistem RFID cu Keypad + Parola + Buzzer ===");

  if (!EEPROM.begin(EEPROM_SIZE)) {
    Serial.println("Eroare la initializarea EEPROM!");
    while (1);
  }

  // Verificam daca parola exista deja
  bool passwordExists = true;
  for (int i = 0; i < PASSWORD_LENGTH; i++) {
    char c = EEPROM.read(PASSWORD_ADDR + i);
    if (c == 0xFF || c == '\0') {
      passwordExists = false;
      break;
    }
  }

  if (!passwordExists) {
    Serial.println("Trebuie sa setezi o parola de 4 cifre.");
    setPassword();
  } else {
    passwordSet = true;
  }

  // Mesaj initial
  Serial.println("\nIntrodu codul de verificare pentru a accesa informatiile cardului.");
}

void loop() {
  // ======== Citire CARD ========
  if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
    Serial.print("Card detectat. UID: ");
    String uid = "";
    for (byte i = 0; i < mfrc522.uid.size; i++) {
      uid += String(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
      uid += String(mfrc522.uid.uidByte[i], HEX);
    }
    uid.toUpperCase();
    Serial.println(uid);

    for (int i = 0; i < uid.length(); i++) {
      EEPROM.write(i, uid[i]);
    }
    EEPROM.write(uid.length(), '\0'); 
    EEPROM.commit();
    Serial.println("UID salvat in EEPROM!");
    delay(1000);
  }

  // ======== Citire Tastatura ========
  char key = keypad.getKey();
  if (key) {
    Serial.print("Tasta apasata: ");
    Serial.println(key);

    // Verificare pentru resetare parola cu 2 apasari ale tastei 0
    if (key == '0') {
      unsigned long currentTime = millis();
      if (zeroPressedOnce && (currentTime - lastZeroPress < 1000)) { // 2 apăsări în 1 sec
        Serial.println("\nActivat modul de setare a noii parole!");
        setPassword();
        zeroPressedOnce = false;
      } else {
        zeroPressedOnce = true;
        lastZeroPress = currentTime;
      }
      return; // iesim din loop-ul tastelor pentru a nu verifica parola imediat
    } else {
      zeroPressedOnce = false; // reset dacă e altă tastă
    }

    // Daca apesi D, cerem parola
    if (key == 'D') {
      if (verifyPassword()) {
        Serial.print("Cod/UID din EEPROM: ");
        int i = 0;
        char c;
        while ((c = EEPROM.read(i)) != '\0' && i < EEPROM_SIZE) {
          Serial.print(c);
          i++;
        }
        Serial.println();
        beepCorrect();
      } else {
        Serial.println("Parola incorecta!");
        beepWrong();
      }
    }
  }
}

///////////////////////
// Functii Parola //
///////////////////////
void setPassword() {
  Serial.println("Introdu o parola noua (4 cifre) si confirma cu #:");
  enteredPassword = "";
  while (enteredPassword.length() < PASSWORD_LENGTH) {
    char key = keypad.getKey();
    if (key) {
      if (key >= '0' && key <= '9') {
        enteredPassword += key;
        Serial.print("*");
      } else if (key == '#') {
        if (enteredPassword.length() == PASSWORD_LENGTH) break;
      }
    }
  }

  for (int i = 0; i < PASSWORD_LENGTH; i++) {
    EEPROM.write(PASSWORD_ADDR + i, enteredPassword[i]);
  }
  EEPROM.commit();
  passwordSet = true;
  Serial.println("\nParola setata cu succes!");
}

bool verifyPassword() {
  if (!passwordSet) return false;

  Serial.println("\nIntrodu parola pentru acces si confirma cu #: ");
  enteredPassword = "";
  while (true) {
    char key = keypad.getKey();
    if (key) {
      if (key >= '0' && key <= '9') {
        enteredPassword += key;
        Serial.print("*");
      } else if (key == '#') {
        if (enteredPassword.length() == PASSWORD_LENGTH) {
          Serial.println();
          break;
        }
      }
    }
  }

  for (int i = 0; i < PASSWORD_LENGTH; i++) {
    if (EEPROM.read(PASSWORD_ADDR + i) != enteredPassword[i]) {
      return false;
    }
  }
  return true;
}

///////////////////////
// Functii Buzzer //
///////////////////////
void beepCorrect() {
  digitalWrite(BUZZER_PIN, HIGH);
  delay(200);
  digitalWrite(BUZZER_PIN, LOW);
}

void beepWrong() {
  for (int i = 0; i < 2; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(200);
    digitalWrite(BUZZER_PIN, LOW);
    delay(200);
  }
}
