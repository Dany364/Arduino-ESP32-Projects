#include <MFRC522.h>

#define SS_PIN 21
#define RST_PIN 22
#define BUZZER 15

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(115200);
  SPI.begin(18, 19, 23, 21);
  mfrc522.PCD_Init();
  pinMode(BUZZER, OUTPUT);
  Serial.println("Test RFID - Apropie un card...");
}

void loop() {
  // Citire card
  if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
    Serial.print("UID: ");
    for (byte i = 0; i < mfrc522.uid.size; i++) {
      Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ");
      Serial.print(mfrc522.uid.uidByte[i], HEX);
    }
    Serial.println();
    delay(1000); // mic delay
  }

  // Verificare comenzi din Python
  if (Serial.available() > 0) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();
    if (cmd == "BUZZ") {
      digitalWrite(BUZZER, HIGH);
      delay(200);
      digitalWrite(BUZZER, LOW);
    }
  }
}

UID_VALID = "40 0A 7A 14"

print("Aștept taguri RFID...")

while True:
    if ser.in_waiting > 0:
        line = ser.readline().decode("utf-8").strip()
        if line.startswith("UID:"):
            uid = line.split("UID:")[1].strip()
            print(f"Am citit cardul cu UID: {uid}")

            if uid.upper() == UID_VALID:  # comparare fără probleme majuscule/minuscule
                print("✅ Card valid! Pornește buzzerul.")
                ser.write(b"BUZZ\n")
            else:
                print("❌ Card invalid!")
